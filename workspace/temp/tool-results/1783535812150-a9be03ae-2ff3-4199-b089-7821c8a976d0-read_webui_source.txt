64: // claude-haiku-4-5-20251001 → "Claude Haiku 4.5"
65: // gpt-5.5 → "GPT 5.5"   ·   grok-4.20-reasoning → "Grok 4.20"
66: export function prettifyModelName(model, provider) {
67:   const raw = String(model || '').trim();
68:   const providerId = String(provider || '').trim().toLowerCase();
69:   if (!raw) return _providerLabel(providerId) || 'Model';
70:   let s = raw.toLowerCase();
71:   // Drop date stamps like -20251001 / 20250514.
72:   s = s.replace(/-?20\d{6}\b/g, '');
73:   // Drop trailing qualifier words that add noise on a tiny badge.
74:   s = s.replace(/-(reasoning|non-reasoning|latest|preview|exp|instruct|thinking|online)\b/g, '');
75:   s = s.replace(/-\d{4}\b/g, '');
76:   // Strip the redundant "claude-" family prefix (anthropic).
77:   s = s.replace(/^claude-/, '');
78:   // Version dashes between digits become dots: 4-5 → 4.5.
79:   s = s.replace(/(\d)-(\d)/g, '$1.$2');
80:   // Remaining separators → spaces.
81:   s = s.replace(/[-_]+/g, ' ').replace(/\s+/g, ' ').trim();
82:   if (!s) s = raw;
83: 
84:   if (/^gpt\b/.test(s)) {
85:     s = s.replace(/^gpt\b/i, 'GPT');
86:   } else if (/^o\d\b/.test(s)) {
87:     s = s.toUpperCase();
88:   } else if (providerId === 'anthropic' || /^(opus|sonnet|haiku)\b/.test(s)) {
89:     s = s.replace(/^(opus|sonnet|haiku)\b/i, 'Claude $1');
90:   } else if (providerId === 'gemini' || /^gemini\b/.test(s)) {
91:     s = s.replace(/^gemini\b/i, 'Gemini');
92:   } else if (providerId === 'xai' || /^grok\b/.test(s)) {
93:     s = s.replace(/^grok\b/i, 'Grok');
94:   } else if (providerId === 'perplexity' || /^sonar\b/.test(s)) {
95:     s = s.replace(/^sonar\b/i, 'Sonar');
96:   } else {
97:     s = s.replace(/\b[a-z]/g, (ch) => ch.toUpperCase());
98:   }
99: 
100:   s = s.replace(/\bmini\b/gi, 'mini')
101:     .replace(/\b(Pro|Flash|Lite|Build|Codex|Max|Haiku|Opus|Sonnet|Deep|Research|Multi|Agent)\b/gi, (word) => word.charAt(0).toUpperCase() + word.slice(1).toLowerCase())
102:     .replace(/\s+/g, ' ')
103:     .trim();
104:   if (s.length > 24) s = s.slice(0, 23).trim() + '…';
105:   return s;
106: }

131: // `<input switch>` overlay sits on top of the button so the user's physical tap

456:   ids.sort((a, b) => {
457:     const ia = order.indexOf(a); const ib = order.indexOf(b);
458:     return (ia < 0 ? 99 : ia) - (ib < 0 ? 99 : ib);
459:   });
460: 
461:   if (!ids.length) {
462:     _setSheetBody('<div class="pm-msheet-empty">No providers with saved credentials. Add an API key or connect a provider in Settings.</div>');
463:     return;
464:   }
465: 
466:   const rows = ids.map((id) => `
467:     <button type="button" class="pm-msheet-row" data-provider="${_esc(id)}">
468:       <span class="pm-msheet-row-label">${_esc(_providerLabel(id))}</span>
469:       ${id === activeProvider ? '<span class="pm-msheet-dot" title="Current"></span>' : ''}
470:       <span class="pm-msheet-chev">›</span>
471:     </button>`).join('');
472:   const body = _setSheetBody(`<div class="pm-msheet-rows">${rows}</div>`);
473:   if (!body) return;
474:   body.querySelectorAll('[data-provider]').forEach((btn) => {
475:     btn.addEventListener('click', () => _renderModelList(btn.getAttribute('data-provider')));
476:   });

479: function _renderModelList(provider) {

495:   document.getElementById('pm-msheet-back')?.addEventListener('click', _renderProviderList);
496:   if (!body) return;
497:   body.querySelectorAll('[data-model]').forEach((btn) => {

502: async function _switchModel(provider, model) {
503:   if (!provider || !model) return;
504:   try {
505:     await mobileGatewayFetch('/api/settings/model', {
506:       method: 'POST',
507:       body: JSON.stringify({ provider, model }),
508:     });
509:     _llmCache = null; // force refresh
510:     _toast(`Model → ${prettifyModelName(model, provider)}`, 'success');
511:     _closeSheet();
512:     await refreshMobileModelBadge(true);
513:     try { window.dispatchEvent(new CustomEvent('pm-model-changed', { detail: { provider, model } })); } catch {}
514:   } catch (err) {
515:     _toast(err?.message || 'Could not switch model', 'error');
516:   }
517: }

519: // ── Press / long-press gesture wiring (delegated, attached once) ─────────────
520: let _wired = false;
521: const LONG_PRESS_MS = 480;
522: const MOVE_CANCEL_PX = 12;
523: 
524: export function initMobileModelBadge() {
525:   if (_wired) return;
526:   _wired = true;
527: 
528:   let pressTimer = null;
529:   let longFired = false;
530:   let startX = 0;
531:   let startY = 0;
532:   let pressBadge = null;
533:   let suppressNextClick = false;
534: 
535:   const findBadge = (target) => (target?.closest ? target.closest('.pm-model-badge') : null);
536:   const setSuppressNativeSelection = (on) => {
537:     document.documentElement.classList.toggle('pm-model-badge-pressing', !!on);
538:     document.body?.classList?.toggle('pm-model-badge-pressing', !!on);
539:   };
540: 
541:   document.addEventListener('pointerdown', (e) => {
542:     const badge = findBadge(e.target);
543:     if (!badge) return;
544:     e.preventDefault();
545:     longFired = false;
546:     pressBadge = badge;
547:     setSuppressNativeSelection(true);
548:     startX = e.clientX; startY = e.clientY;
549:     if (pressTimer) clearTimeout(pressTimer);
550:     pressTimer = setTimeout(() => {
551:       pressTimer = null;
552:       longFired = true;
553:       pmHaptic(18);
554:       _openSwitchSheet();
555:     }, LONG_PRESS_MS);
556:   });
557: 
558:   document.addEventListener('pointermove', (e) => {
559:     if (!pressTimer) return;
560:     if (Math.abs(e.clientX - startX) > MOVE_CANCEL_PX || Math.abs(e.clientY - startY) > MOVE_CANCEL_PX) {
561:       clearTimeout(pressTimer); pressTimer = null; pressBadge = null; setSuppressNativeSelection(false);
562:     }
563:   });
564: 
565:   const cancel = () => {
566:     if (pressTimer) { clearTimeout(pressTimer); pressTimer = null; }
567:     pressBadge = null;
568:     setSuppressNativeSelection(false);
569:   };
570:   document.addEventListener('pointerup', (e) => {
571:     const badge = pressBadge;
572:     const wasLong = longFired;
573:     const moved = Math.abs(e.clientX - startX) > MOVE_CANCEL_PX || Math.abs(e.clientY - startY) > MOVE_CANCEL_PX;
574:     cancel();
575:     if (badge && !wasLong && !moved) {
576:       suppressNextClick = true;
577:       _openReasoningSheet();
578:       setTimeout(() => { suppressNextClick = false; }, 350);
579:     }
580:   });
581:   document.addEventListener('pointercancel', cancel);
582:   document.addEventListener('selectstart', (e) => {
583:     if (pressBadge || findBadge(e.target)) {
584:       e.preventDefault();
585:       e.stopPropagation();
586:     }
587:   }, true);
588:   document.addEventListener('contextmenu', (e) => {
589:     if (pressBadge || findBadge(e.target)) {
590:       e.preventDefault();
591:       e.stopPropagation();
592:     }
593:   }, true);
594: 
595:   document.addEventListener('click', (e) => {
596:     const badge = findBadge(e.target);
597:     if (!badge) return;
598:     if (suppressNextClick) { e.preventDefault(); e.stopPropagation(); return; }
599:     if (longFired) { e.preventDefault(); e.stopPropagation(); longFired = false; return; }
600:     _openReasoningSheet();
601:   });
602: 
603:   // Keep the label fresh on navigation and when the model changes elsewhere.
604:   window.addEventListener('hashchange', () => { refreshMobileModelBadge(false).catch(() => {}); });
605:   window.addEventListener('pm-model-changed', (event) => {
606:     const detail = event?.detail || {};
607:     refreshMobileModelBadge(true, detail).catch(() => {});
608:   });
609: 
610:   refreshMobileModelBadge(true).catch(() => {});
611: }
